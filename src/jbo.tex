\tran{la lojban. (ISO 639-3 jbo)}{le lojbo be la .ionedas.}{
	ganai la'o zoi $\mathcal C$ zoi cmalu klesi gi la'o zoi $F$ zoi funkto lo klesi lo klesi be lo'i cmima. .i ro dacti be la'o zoi $X$ zoi be la'o zoi $\mathcal C$ zoi ku'o zo'u rarna bijekci cu zasti fa'u le cmima be le rarna cenba be la'o zoi $\hom(-,X)\Rightarrow F$ zoi bei le cmima be la'o zoi $FX$ zoi .i le bi'ekci cu se ciksi le mapti
	\[
		\Big(\xi : \hom(-,X)\Rightarrow F\Big) \mapsto \xi_X(1_X)
	\]
	(le fancu cu bijektive).
}
